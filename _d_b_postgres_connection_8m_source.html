126 - (NSArray *)executeSQL:(NSString *)sql substitutions:(id)substitutions error:(NSError **)outErr
127 {
128  DBLog(@"%@ -- %@", sql, substitutions);
129  BOOL isDict = [substitutions isKindOfClass:[NSDictionary class]] ||
130  [substitutions isKindOfClass:[NSMapTable class]];
131  NSParameterAssert(!substitutions ||
132  [substitutions isKindOfClass:[NSArray class]] ||
133  [substitutions isKindOfClass:[NSPointerArray class]] ||
134  isDict);
135  //DBDebugLog(@"Executing SQL: %@ subs: %@", sql, substitutions);
136  // Prepare the query
137 
138  NSUInteger paramCount = [substitutions count];
139  const char *paramValues[paramCount];
140  int paramLengths[paramCount];
141  int paramFormats[paramCount]; // 0 = text, 1 = binary
142 
143  NSString *key;
144  int i = 0;
145  for(__strong id sub in substitutions) {
146  if(isDict) {
147  key = sub;
148  sub = substitutions[key];
149  // Unfortunately Postgres doesn't support `:<param>` style named parameters => we need to manually anonymize
150  NSString *paramToken = [@":" stringByAppendingString:key];
151  NSRange tokenRange = rangeOfParameterToken(paramToken, sql);
152  if(tokenRange.location == NSNotFound)
153  continue;
154  sql = [sql stringByReplacingOccurrencesOfString:paramToken
155  withString:[@"$" stringByAppendingFormat:@"%d", i]];
156  }
157 
158  if([sub isKindOfClass:[NSString class]] || [[sub className] isEqualToString:@"NSCFString"]) {
159  paramValues[i] = [sub UTF8String];
160  paramLengths[i] = [sub length];
161  paramFormats[i] = 0;
162  } else if([sub isKindOfClass:[NSNumber class]] || [sub isKindOfClass:NSClassFromString(@"TQNumber")]) {
163  NSString *str;
164  switch (*[sub objCType]) {
165  case 'd':
166  case 'f':
167  str = [NSString stringWithFormat:@"%f", [sub doubleValue]];
168  break;
169  case 'l':
170  case 'L':
171  str = [NSString stringWithFormat:@"%ld", [sub longValue]];
172  break;
173  case 'q':
174  case 'Q':
175  str = [NSString stringWithFormat:@"%lld", [sub longLongValue]];
176  break;
177  case 'B': // C++/C99 bool
178  case 'c': // ObjC BOOL
179  str = [NSString stringWithFormat:@"%d", [sub intValue]];
180  break;
181  default:
182  str = [NSString stringWithFormat:@"%ld", [sub longValue]];
183  break;
184  }
185  paramValues[i] = [str UTF8String];
186  paramLengths[i] = [str length];
187  paramFormats[i] = 0;
188  } else if([sub isKindOfClass:[NSData class]]) {
189  paramValues[i] = [sub bytes];
190  paramLengths[i] = [sub length];
191  paramFormats[i] = 1;
192  } else if([sub isKindOfClass:[NSDate class]]) {
193  NSString *timestamp = NSDateToPostgresTimestamp(sub);
194  paramValues[i] = [timestamp UTF8String];
195  paramLengths[i] = [timestamp length];
196  paramFormats[i] = 0;
197  } else if(!sub || [sub isMemberOfClass:[NSNull class]]) {
198  paramValues[i] = NULL;
199  paramLengths[i] = 0;
200  paramFormats[i] = 0;
201  } else
202  [NSException raise:@"Unrecognized object type"
203  format:@"DatabaseKit doesn't know how to handle this type of object: %@ class: %@", sub, [sub className]];
204  ++i;
205  }
206 
207  PGresult *result = PQexecParams(_connection, [sql UTF8String], paramCount, NULL, paramValues, paramLengths, paramFormats, 0);
208  switch (PQresultStatus(result)) {
209  case PGRES_BAD_RESPONSE:
210  case PGRES_FATAL_ERROR:
211  DBDebugLog(@"Query error: %@", [NSString stringWithUTF8String:PQresultErrorMessage(result)]);
212  if(outErr)
213  *outErr = [[NSError alloc] initWithDomain:DBConnectionErrorDomain
214  code:DBPostgresQueryFailed
215  userInfo:@{ NSLocalizedDescriptionKey: @(PQresultErrorMessage(result)) }];
216  return nil;
217  default:
218  break;
219  }
220  int rowCount = PQntuples(result);
221  NSArray *columnNames = [self columnsForResult:result];
222  NSMutableArray *rowArray = [NSMutableArray array];
223  NSMutableDictionary *columns;
224  for(int i = 0; i < rowCount; ++i) {
225  // construct the dictionary for the row
226  columns = [NSMutableDictionary dictionary];
227  int j = 0;
228  for(NSString *columnName in columnNames) {
229  columns[columnName] = [self valueForRow:i column:j result:result];
230  ++j;
231  }
232  [rowArray addObject:columns];
233  }
234  PQclear(result);
235  return rowArray;
236 }

257 - (id)valueForRow:(unsigned int)rowIndex column:(unsigned int)colIndex result:(PGresult *)result
258 {
259  const char *bytes = PQgetvalue(result, rowIndex, colIndex);
260  if(!bytes)
261  return [NSNull null];
262 
263  NSUInteger length = PQgetlength(result, rowIndex, colIndex);
264  switch(PQftype(result, colIndex))
265  {
266  case INT2OID:
267  return @(atoi(bytes));
268  case INT4OID:
269  return @(strtoul(bytes, (char **)NULL, 10));
270  case INT8OID:
271  return @(strtoull(bytes, (char **)NULL, 10));
272  case FLOAT4OID:
273  case FLOAT8OID:
274  return @(atof(bytes));
275  break;
276  case DATEOID:
277  case TIMEOID:
278  case TIMESTAMPOID:
279  case TIMESTAMPTZOID:
280  return NSDateFromPostgresTimestamp(@(bytes));
281  case BYTEAOID: {
282  const unsigned char *rawBytes = PQunescapeBytea((const unsigned char *)bytes, &length);
283  NSData *data = [NSData dataWithBytes:rawBytes length:length];
284  PQfreemem((void *)rawBytes);
285  return data;
286  }
287  case VARCHAROID:
288  case TEXTOID:
289  default:
290  return @(bytes);
291  break;
292  }
293  return nil;
294 }

341 static NSRange rangeOfParameterToken(NSString *param, NSString *query)
342 {
343  BOOL inDblQuotes = NO, inSingleQuotes = NO, escapeNext = NO;
344  const char *needle = [param UTF8String], *haystack = [query UTF8String];
345  NSUInteger needleLen = [param length];
346 
347  for(int i = 0; i < [query length]; ++i) {
348  if(escapeNext)
349  escapeNext = NO;
350  else if(haystack[i] == '"' && !inSingleQuotes)
351  inDblQuotes = !inDblQuotes;
352  else if(haystack[i] == '\'' && !inDblQuotes)
353  inSingleQuotes = !inSingleQuotes;
354  else if(haystack[i] == '\\')
355  escapeNext = YES;
356  else if(inSingleQuotes || inDblQuotes)
357  continue;
358  else if(strncmp(haystack+i, needle, needleLen) == 0)
359  return (NSRange){ i, needleLen };
360  }
361  return (NSRange){ NSNotFound, 0 };
362 }
